자바스크립트 VS 제이쿼리 문법 비교

제이쿼리 문법과 자바스크립트 문법을 비교합니다.

01. 문서 로딩 : script가 문서 위에 있는 경우

    window.onload = function(){
    //실행 코드
    };  

    $(document).ready(function(){
    //실행코드
    });

02. id 선택자

    document.getElementById("id").style.color = "purple";

    $("#id").css("color","purple");

03. class 선택자

    document.querySelector(".class") .style.color = "purple";

    $(".class").css("color","purple");

04. 다중 선택자

    document.querySelectorAll(".class") .style.color = "purple";

    $(".class").css("color","purple");

05. 자식 선택자

    document.querySelectorAll(".class .child") .style.color = "purple";

    $(".class .child").css("color","purple");

01. 스타일

document.querySelector(".class").style.color = "purple";
    document.querySelector(".class").style.transform = "translate(100px)";

    $(".class").css("color","purple");
    $(".class").css("transform","translate(100px)");

01. 클래스 추가

    document.querySelector(".class").classList.add("show");

    $(".class").addClass("show");

02. 클래스 제거

    document.querySelector(".class").classList.remove("show");

    $(".class").removeClass("show");

03. 클래스 토글

    document.querySelector(".class").classList.toggle("show");

    $(".class").toggleClass("show");

04. 클래스 확인

    document.querySelector(".class").contain("show");

    $(".class").hasClass("show");

01. 텍스트 삽입

    document.querySelector(".class1").innerText = "1글자를 추가합니다.";   
    document.querySelector(".class2").textContent = "2글자를 추가합니다.";

    $(".class").hasClass("show");

01. 태그 & 텍스트 삽입

    document.querySelector(".class").contain("show");

    $(".class").hasClass("show");

01. show

    document.querySelector(".class").style.display = "block"

    $(".class").show();

02. hide

    document.querySelector(".class").style.display = "none"

    $(".class").hide();

03. fadeIn

    document.querySelector(".class").addEventListener("click",function(){
    document.querySelector(".class").style.opacity = "1"
    document.querySelector(".class").style.transition = "400ms"
    });

    $(".class").fadeIn();

04. fadeOut

    document.querySelector(".class").addEventListener("click",function(){
    document.querySelector(".class").style.opacity = "0"
    document.querySelector(".class").style.transition = "400ms"
    });

    $(".class").fadeOut();

05. slideUp

    document.querySelector(".class").addEventListener("click",function(){
    document.querySelector(".class").style.height = "0"
    document.querySelector(".class").style.transition = "400ms"
    });

    $(".class").slideUp();

05. slideDown

    document.querySelector(".class").addEventListener("click",function(){
    document.querySelector(".class").style.height = "100px"
    document.querySelector(".class").style.transition = "400ms"
    });

    $(".class").slideDown();

01. width

    document.querySelector(".class").clientWidth;  //padding 포함
    document.querySelector(".class").offsetWidth;  //padding 포함
    document.querySelector(".class").getBoundingClientRect() ;

    $(".class").width();
    $(".class").innerWidth();  //padding 포함
    $(".class").outerWidth();  //padding,border 포함

02. height

    document.querySelector(".class").clientHeight; //padding 포함
    document.querySelector(".class").offsetHeight; //padding 포함
    document.querySelector(".class").getBoundingClientRect();

    $(".class").height();
    $(".class").innerHeight(); //padding 포함
    $(".class").outerHeight(); //padding,border 포함

01. clone

    document.querySelector(".class").cloneNode(true);

    $(".class").clone();

01. before

    let clone = document.querySelector(".class").cloneNode(true);
    document.querySelector(".class").insertAdjacentElement("beforebegin",clone);

    let clone = $(".class").clone();
    $(".class").before(clone);

02. after

    let clone = document.querySelector(".class").cloneNode(true);
    document.querySelector(".class").insertAdjacentElement("afterend",clone);

    let clone = $(".class").clone();
    $(".class").after(clone);

03. prepend

    let clone = document.querySelector(".class").cloneNode(true);
    document.querySelector(".class").insertAdjacentElement("afterbegin",clone);

    let clone = $(".class").clone();
    $(".class").prepend(clone);

04. append

    let clone = document.querySelector(".class").cloneNode(true);
    document.querySelector(".class").insertAdjacentElement("beforeend",clone);

    let clone = $(".class").clone();
    $(".class").append(clone);
